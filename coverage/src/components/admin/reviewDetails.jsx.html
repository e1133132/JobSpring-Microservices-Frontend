function formatDate(iso) {
    try {
        if (!iso) return "-";
        const d = new Date(iso);
        return isNaN(d) ? "-" : d.toLocaleString();
    } catch {
        return "-";
    }
}
 
function buildFileUrl(url) {
    if (!url) return "";
    if (/^https?:\/\//i.test(url)) return url;
    return `${window.location.origin}${url.startsWith("/") ? "" : "/"}${url}`;
}
